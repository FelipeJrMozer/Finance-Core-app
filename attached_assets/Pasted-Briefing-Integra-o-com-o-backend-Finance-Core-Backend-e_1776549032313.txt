2) Registrar Push Token no backend (CRÍTICO)
Arquivo: artifacts/finance-core/services/NotificationService.native.ts — adicionar no final:

import Constants from 'expo-constants';
import { apiPost } from './api';
export async function registerPushTokenWithBackend(): Promise<void> {
  if (Platform.OS === 'web' || !Device.isDevice) return;
  try {
    const projectId =
      Constants.expoConfig?.extra?.eas?.projectId ??
      (Constants.easConfig as any)?.projectId;
    const tokenData = await Notifications.getExpoPushTokenAsync(
      projectId ? { projectId } : undefined
    );
    const token = tokenData.data;
    if (!token) return;
    await apiPost('/api/devices/register', {
      token,
      platform: Platform.OS, // 'ios' | 'android'
    });
  } catch (err) {
    console.warn('[Push] Falha ao registrar token:', err);
  }
}

Arquivo: artifacts/finance-core/hooks/useNotifications.native.ts — chamar após permissão concedida (dentro de scheduleAll, logo depois de if (!granted) return;):
