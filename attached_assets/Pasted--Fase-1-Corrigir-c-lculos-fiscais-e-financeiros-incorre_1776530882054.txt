10. parseFloat e split('T') quebram em formatos inesperados
Arquivo: artifacts/finance-core/context/FinanceContext.tsx (linhas ~198 e ~216)
Correção:

Criar helper artifacts/finance-core/utils/parse.ts:
export function parseAmount(raw: unknown): number {
  if (typeof raw === 'number') return Number.isFinite(raw) ? raw : 0;
  if (typeof raw !== 'string') return 0;
  // aceita "1234.56", "1.234,56" e "1234,56"
  const normalized = raw.replace(/\./g, '').replace(',', '.');
  const n = parseFloat(normalized);
  return Number.isFinite(n) ? n : 0;
}
export function parseDate(raw: unknown): string {
  if (typeof raw !== 'string') return new Date().toISOString().slice(0, 10);
  const d = new Date(raw);
  return Number.isFinite(d.getTime()) ? d.toISOString().slice(0, 10) : new Date().toISOString().slice(0, 10);
}

Substituir todos os parseFloat(raw.amount) e d.split('T')[0] por essas helpers.
